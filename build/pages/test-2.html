    <script>
    const accordionButtons = document.querySelectorAll('.accordion-button');
    
    accordionButtons.forEach(button => {
    button.addEventListener('click', () => {
        const content = button.nextElementSibling;
        const icon = button.querySelector('svg');
        const allContents = document.querySelectorAll('.accordion-content');
    
        // Close all other accordion contents and reset their icons
        allContents.forEach(item => {
            if (item !== content && !item.classList.contains('hidden')) {
                item.classList.add('hidden');
                item.previousElementSibling.classList.remove('bg-Neutral-50');
                item.previousElementSibling.classList.add('bg-white');
                item.previousElementSibling.querySelector('svg').classList.remove('rotate-180');
            }
        });
    
        // Toggle visibility of content
        content.classList.toggle('hidden');
    
        // Toggle button background color
        button.classList.toggle('bg-Neutral-50');
        button.classList.toggle('bg-white');
    
        // Rotate icon
        icon.classList.toggle('rotate-180');
    });
    });
    </script>
