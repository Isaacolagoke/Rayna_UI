Rayna is a comprehensive design system and component library built specifically for Figma (V1.0). It provides a set of pre-designed UI elements, components, and guidelines that ensure consistent and efficient design across your projects.
Simply duplicate the library into your Figma projects to start utilizing its design assets and components.
Yes, absolutely! Rayna components are designed with customization in mind. You can easily adjust colors, typography, and other design attributes to align with your brand's unique identity while maintaining the consistency that Rayna offers.
Yes, updates to Rayna will be made in the near future. You'll be notified of any updates, and you can seamlessly integrate the latest versions into your ongoing projects.

<!-- accordion code -->
<script>
const accordionButtons = document.querySelectorAll('.accordion-button');

accordionButtons.forEach(button => {
button.addEventListener('click', () => {
    const content = button.nextElementSibling;
    const icon = button.querySelector('svg');
    const allContents = document.querySelectorAll('.accordion-content');

    // Close all other accordion contents and reset their icons
    allContents.forEach(item => {
        if (item !== content && !item.classList.contains('hidden')) {
            item.classList.add('hidden');
            item.previousElementSibling.classList.remove('bg-Neutral-50');
            item.previousElementSibling.classList.add('bg-white');
            item.previousElementSibling.querySelector('svg').classList.remove('rotate-180');
        }
    });

    // Toggle visibility of content
    content.classList.toggle('hidden');

    // Toggle button background color
    button.classList.toggle('bg-Neutral-50');
    button.classList.toggle('bg-white');

    // Rotate icon
    icon.classList.toggle('rotate-180');
});
});
</script>

<!-- carousel code -->
<script>
// Variables
const slides = document.querySelector('#carousel');
const slideWidth = slides.querySelector('div').offsetWidth;
let currentIndex = 0;
let intervalId;

// Function to scroll left
function scrollLeft() {
if (currentIndex > 0) {
currentIndex--;
slideToIndex();
} else {
currentIndex = slides.children.length - 1;
slides.style.transform = `translateX(-${currentIndex * slideWidth}px)`;
}
}

// Function to scroll right
function scrollRight() {
if (currentIndex < slides.children.length - 1) {
currentIndex++;
slideToIndex();
} else {
currentIndex = 0;
slides.style.transform = 'translateX(0)';
}
}

// Function to slide to a specific index
function slideToIndex() {
slides.style.transform = `translateX(-${currentIndex * slideWidth}px)`;
}

// Function to automatically scroll after 5 seconds
function autoScroll() {
intervalId = setInterval(() => {
if (currentIndex < slides.children.length - 1) {
    currentIndex++;
    slideToIndex();
} else {
    currentIndex = 0;
    slides.style.transform = 'translateX(0)';
}
}, 3000);
}

// Start auto-scrolling
autoScroll();

// Stop auto-scrolling when mouse hovers over the carousel
slides.addEventListener('mouseenter', () => {
clearInterval(intervalId);
});

// Resume auto-scrolling when mouse leaves the carousel
slides.addEventListener('mouseleave', () => {
autoScroll();
});
</script>
